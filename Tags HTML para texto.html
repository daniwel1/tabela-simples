       Código em C# para fazer um programa de impar ou par.
    
    class Program
    {
          static void Main(string[] args)                                                                       
        {
            int num;                                  // Definimos que nossa variavel será um numero inteiro.
            
            Console.WriteLine("Insira um número");    //Recebendo informação do usuário.                                                 
            num = int.Parse(Console.ReadLine());      // Variável declarada como numero inteiro.  
    
            if (num % 2 == 0)                         // O número a ser digitado pelo usuário será dividido por 2 para chegar ao objetivo final. 
            { 
             Console.WriteLine("O número " + num + " é par");
            }
            else
            {
                Console.WriteLine("O número " + num + " é impar");
            }
        }    
    }
